function doPost(e) {
  var d = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.openById(
    d.sheetId || PropertiesService.getScriptProperties()
      .getProperty('SHEET_ID'));
  if (d.action === 'backup') {
    var sh = ss.getSheetByName('backup') || ss.insertSheet('backup');
    sh.clearContents();
    sh.getRange(1,1).setValue(d.payload);
    return ContentService.createTextOutput(
      JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  if (d.action === 'restore') {
    var sh = ss.getSheetByName('backup');
    if (!sh) return ContentService.createTextOutput(
      JSON.stringify({status:'empty'}))
      .setMimeType(ContentService.MimeType.JSON);
    return ContentService.createTextOutput(
      JSON.stringify({status:'ok',
        payload:sh.getRange(1,1).getValue()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
